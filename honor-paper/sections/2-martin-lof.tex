\begin{definition}
    Consider a type family $B$ over $A$.
    The \emph{dependent pair type} (or $\Sigma$-type) is defined to be the inductive type $\Sigma_{(x:A)} B(x)$ equipped with a pairing function
    \[
        (-,-) :\Prod_{(x:A)}  \Paren{B(x)\to\Sum_{(y:A)}B(y)}
    \]
\end{definition}
